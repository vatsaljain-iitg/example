#include < stdio.h >
int main()
{
int n;
scanf("%d", &n); // Task 01
int prices[n];
for (int i = 0; i < n; i++)
scanf("%d", &prices[i]); // Task 02
int max = prices[1] - prices[0];
for (int i = 0; i < n; i++)
for (int j = i + 1; j < n; j++)
if (prices[j] - prices[i] > max)
max = prices[j] - prices[i];
// Task 03
if (max >= 0)
printf("%d\n", max);
else
printf("0\n");
// Task 04
return 0;
}
